\chapter{Geometry of Number Fields}
  Throughout, all rings will be understood to be commutative and with unity. We will also use the theory of integral lattices in \cref{sec:integral_lattices} and some results of the beta function is \cref{sec:the_gamma_function}.
  \section{Minkowski Space}
    Let \(K\) be number field of degree \(n\). If \(\s \in \Hom_{\Q}(K,\conj{\Q})\) then either \(\s\) is real or complex and in the latter case it is has a paired \(\Q\)-embedding \(\conj{\s}\) which is the complex conjugate of \(\s\). Accordingly, let \(r_{1}\) and \(2r_{2}\) be the number of real and complex \(\Q\)-embeddings of \(K\) into \(\conj{\Q}\) respectively. We call the pair \((r_{1},r_{2})\) the \textbf{signature}\index{signature} of \(K\) and it satisfies the relation
    \[
      n = r_{1}+2r_{2}.
    \]
    Setting
    \[
      K_{\C} = \C^{n},
    \]
    makes \(K_{\C}\) into a component-wise \(\C\)-algebra. Moreover, \(K_{\C}\) is also a complex Hilbert space with respect to the standard complex inner product which we denote by \(\<\cdot,\cdot\>_{K_{\C}}\). We also write \(d\l_{K_{\C}}\) for the induced Lebesgue measure on \(K_{\C}\) by this inner product. Also let \(\Trace_{K_{\C}}\) and \(\Norm_{K_{\C}}\) denote the trace and norm of \(K_{\C}/\C\). Because \(K_{\C}\) is a component-wise \(\C\)-algebra, the trace and the norm are given by
    \[
      \Trace_{K_{\C}}(\mathbf{z}) = \sum_{\s}z_{\s} \quad \text{and} \quad \Norm_{K_{\C}}(\mathbf{z}) = \prod_{\s}z_{\s},
    \]
    for all \(\mathbf{z} \in K_{\R}\). We define the \textbf{canonical embedding}\index{canonical embedding} \(j\) of \(K\) to be the \(\Q\)-embedding
    \[
      j:K \to K_{\C} \qquad \k \mapsto (\s(\k))_{\s}.
    \]
    In other words, the canonical embedding is a component-wise \(\Q\)-embedding of \(K\) into \(K_{\C}\) via the \(\s\). In view of \cref{prop:formulas_for_trace_and_norm}, we have
    \[
      \Trace_{K_{\C}}(j(\k)) = \Trace_{K}(\k) \quad \text{and} \quad \Norm_{K_{\C}}(j(\k)) = \Norm_{K}(\k),
    \]
    which is to say that the compositions of the trace and norm of \(K_{\C}/\C\) with the canonical embedding produces the field trace and field norm. Our primary aim is to construct an \(\R\)-subalgebra of \(K_{\C}\) which corrects for the fact that the complex \(\Q\)-embeddings come in pairs. To this end, consider the complex conjugation map
    \[
      F:\C \to \C \qquad z \mapsto \conj{z}.
    \]
    This induces an automorphism of \(K_{\C}\) given by
    \[
      F:K_{\C} \to K_{\C} \qquad \mathbf{z} \mapsto (\conj{z_{\conj{\s}}})_{\s},
    \]
    which is an involution. A short computation shows
    \[
      \<F(\mathbf{z}),F(\mathbf{w})\>_{K_{\C}} = F(\<\mathbf{z},\mathbf{w}\>_{K_{\C}}),
    \]
    for all \(\mathbf{z},\mathbf{w} \in K_{\C}\). In other words, inner product \(\<\cdot,\cdot\>_{K_{\C}}\) is \(F\)-equivariant. We define the \textbf{Minkowski space}\index{Minkowski space} \(K_{\R}\) of \(K\) by
    \[
      K_{\R} = \{\mathbf{z} \in K_{\C}:F(\mathbf{z}) = \mathbf{z}\}.
    \]
    In other words, \(K_{\R}\) consists of all of the \(F\)-invariant points of \(K_{\C}\). Note that \(K_{\R}\) is an \(\R\)-subalgebra of \(K_{\C}\). We denote the restriction of the inner product \(\<\cdot,\cdot\>_{K_{\C}}\) to \(K_{\R}\) by \(\<\cdot,\cdot\>_{K_{\R}}\). Moreover, the inner product \(\<\cdot,\cdot\>_{K_{\R}}\) turns \(K_{\R}\) into a real Hilbert space. Indeed, the conjugate symmetry and \(F\)-equivariance of the inner product together give
    \[
      \conj{\<\mathbf{z},\mathbf{w}\>} = \<\mathbf{z},\mathbf{w}\>,
    \]
    for any \(\mathbf{z},\mathbf{w} \in K_{\R}\). Accordingly, we call \(\<\cdot,\cdot\>_{K_{\R}}\) the \textbf{Minkowski inner product}\index{Minkowski inner product} of \(K\). We denote the restriction of the Lebesgue measure \(d\l_{\C}\) to \(K_{\R}\) by \(d\l_{K_{\R}}\) which is also the Lebesgue measure associated to \(\<\cdot,\cdot\>_{K_{\R}}\). We call \(d\l_{K_{\R}}\) the \textbf{Minkowski measure}\index{Minkowski measure}. Lastly, we denote the restrictions of \(\Trace_{K_{\C}}\) and \(\Norm_{K_{\C}}\) to \(K_{\R}\) by \(\Trace_{K_{\R}}\) and \(\Norm_{K_{\R}}\) respectively which are the norm and trace of \(K_{\R}/\R\). We call \(\Trace_{K_{\R}}\) and \(\Norm_{K_{\R}}\) the \textbf{Minkowski trace}\index{Minkowski trace} and \textbf{Minkowski norm}\index{Minkowski norm} of \(K\) respectively. As the complex \(\Q\)-embeddings of \(K\) into \(\conj{\Q}\) come in conjugate pairs, \(j(K) \subset K_{\R}\) so that the canonical embedding is a \(\Q\)-embedding of \(K\) into Minkowski space. Whence
    \[
      \Trace_{K_{\R}}(j(\k)) = \Trace_{K}(\k) \quad \text{and} \quad \Norm_{K_{\R}}(j(\k)) = \Norm_{K}(\k).
    \]
    So the compositions of the Minkowski trace and Minkowski norm with the canonical embedding produces the field trace and field norm.
    
    To give a more explicit description of Minkowski space, let \(N_{\s} = 1,2\) according to if \(\s\) is real or complex respectively. Also let \(\rho\) and \(\tau\) run over the real \(\Q\)-embeddings and a complete set of representatives for the pairs of complex \(\Q\)-embeddings of \(\Hom_{\Q}(K,\conj{\Q})\) respectively. Note that \(N_{\rho} = 1\) and \(N_{\tau} = 2\). Then
    \[
      K_{\R} = \{\mathbf{z} \in K_{\C}:\Im(z_{\rho}) = 0 \text{ and } z_{\conj{\tau}} = \conj{z_{\tau}} \text{ for all } \rho \text{ and } \tau\}.
    \]
    In fact, this gives an explicit isomorphism between Minkowski space and \(\R^{n}\).
    
    \begin{proposition}\label{prop:Minkowski_space_isomorphism}
      Let \(K\) be a number field of degree \(n\) and signature \((r_{1},r_{2})\). Then the map
      \[
        z_{\s} \mapsto x_{\s} = \begin{cases} z_{\s} & \text{if \(\s = \rho\)}, \\ \Re(z_{\s}) & \text{if \(\s = \tau\)}, \\ \Im(z_{\s}) & \text{if \(\s = \conj{\tau}\)}, \end{cases}
      \]
      induces a component-wise isomorphism between \(K_{\R}\) and \(\R^{n}\) making \(K_{\R}\) a real vector space of dimension \(n\). Moreover, the inner product \(\<\cdot,\cdot\>\) on \(\R^{n}\) induced by the Minkowski inner product is given by
      \[
        \<\mathbf{x},\mathbf{x}'\> = \sum_{\s}N_{\s}x_{\s}x'_{\s},
      \]
      for any \(\mathbf{x},\mathbf{x}' \in \R^{n}\).
    \end{proposition}
    \begin{proof}
      This map has inverse
      \[
        x_{\s} \mapsto z_{\s} = \begin{cases} x_{\s} & \text{if \(\s = \rho\)}, \\ x_{\s}+ix_{\conj{\s}} & \text{if \(\s = \tau\)}, \\ x_{\conj{\s}}+ix_{\s} & \text{if \(\s = \conj{\tau}\)}. \end{cases}
      \]
      As both maps are \(\R\)-linear, it follows that the original map induces a component-wise isomorphism between \(K_{\R}\) and \(\R^{n}\) proving the first statement. For the second statement, let \(\mathbf{z},\mathbf{z}' \in K_{\R}\) and let \(\mathbf{x},\mathbf{x}' \in \R^{n}\) be the corresponding elements in \(\R^{n}\) under this isomorphism respectively. If \(\s = \rho\) then
      \[
        x_{\rho} = z_{\rho} \quad \text{and} \quad x'_{\rho} = z'_{\rho},
      \]
      and thus
      \[
        x_{\rho}x'_{\rho} = z_{\rho}\conj{z'_{\rho}}.
      \]
      If \(\s = \tau\) then
      \[
        x_{\tau} = \Re(z_{\tau}), \quad x_{\conj{\tau}} = \Im(z_{\tau}), \quad x'_{\tau} = \Re(z'_{\tau}), \quad \text{and} \quad x'_{\conj{\tau}} = \Im(z'_{\tau}).
      \]
      Whence a short computation shows
      \[
        2(x_{\tau}x'_{\tau}+x_{\conj{\tau}}x'_{\conj{\tau}}) = z_{\tau}\conj{z'_{\tau}}+z_{\conj{\tau}}\conj{z'_{\conj{\tau}}}.
      \]
      The claim about the inner product follows.
    \end{proof}

    In view of this result, we define the \textbf{Minkowski embedding}\index{Minkowski embedding} \(\s_{K}\) of \(K\) by
    \[
      \s_{K}:K \to \R^{n} \qquad \k \mapsto ((\rho(\k))_{\rho},(\Re(\tau(\k)),\Im(\tau(\k)))_{\tau}).
    \]
    Note that the Minkowski embedding is a \(\Q\)-embedding of \(K\) into \(\R^{n}\) as it is the composition of the canonical embedding and the isomorphism established in the previous result. Also, the Minkowski embedding is independent of the representatives \(\tau\) because they occur in conjugate pairs.
    
    We have now constructed Minkowski space and determined its dimension as a real vector space. Our goal now is to pass from \(\Z\)-lattices of \(K\) to integral lattices of Minkowski space. Being an inner product space, this will permit us to measure fractional ideals of \(K\) geometrically via their images inside Minkowski space. As \(j\) is a \(\Q\)-embedding and any fractional ideal \(\mf{f}\) of \(K\) is a complete \(\Z\)-lattice of \(K\), \(j(\mf{f})\) is a complete integral lattice of \(K_{\R}\). The covolume \(V_{j(\mf{f})}\) of \(j(\mf{f})\) is easily determined.

    \begin{proposition}\label{prop:covolume_of_fractional_ideal_under_j}
      Let \(K\) be a number field with signature \((r_{1},r_{2})\) and let \(\mf{f}\) be a fractional ideal of \(K\). Then \(V_{j(\mf{f})}\) is the absolute value of the determinant of any embedding matrix of a basis for \(K/\Q\) contained in \(\mf{f}\). In particular,
      \[
        V_{j(\mf{f})} = \Norm_{K}(\mf{f})\sqrt{|\D_{K}|},
      \]
      and
      \[
        V_{j(\mc{O}_{K})} = \sqrt{|\D_{K}|}.
      \]
    \end{proposition}
    \begin{proof}
      Upon writing \(\mf{f} = \frac{1}{d}\mf{a}\) for some nonzero \(d \in \Z\) and integral ideal \(\mf{a}\) and \cref{prop:norm_and_field_norms_are_identical}, it suffices to prove the statements for the integral ideal \(\mf{a}\). Let \(\k_{1},\ldots,\k_{n}\) be a basis for \(K/\Q\) contained in \(\mf{a}\). Then the associated generator matrix for \(j(\mf{a})\) is the embedding matrix \(M(\k_{1},\ldots,\k_{n})\). Hence
      \[
        V_{j(\mf{a})} = |\det(M(\k_{1},\ldots,\k_{n}))|,
      \]
      proving the first statement. To prove the first identity of the second statement, in view of \cref{equ:embedding_matrix_discriminant_relation} it suffices to show
      \[
        |\det(M(\k_{1},\ldots,\k_{n}))| = \Norm_{K}(\mf{a})|\det(M(\a_{1},\ldots,\a_{n}))|,
      \]
      for any integral basis \(\a_{1},\ldots,\a_{n}\) for \(K/\Q\). Note that \(\k_{1},\ldots,\k_{n}\) is necessarily a basis for \(\mf{a}\) as a free \(\Z\)-module by \cref{thm:integral_basis_AKBL} and write
      \[
        \k_{i} = \sum_{j}a_{i,j}\a_{j},
      \]
      with \(a_{i,j} \in \Z\). Then \((a_{i,j})\) is the base change matrix from \(\a_{1},\ldots,\a_{n}\) to \(\k_{1},\ldots,\k_{n}\). Whence the image of \(\mc{O}_{K}\) under this matrix is \(\mf{a}\). As \(\Z\) is a principal ideal domain, the smith normal form of this matrix implies that
      \[
        \Norm_{K}(\mf{a}) = |\det((a_{i,j}))|.
      \]
      This fact together with \cref{equ:base_change_discriminant,disc_as_square_of_embedding_matrix} gives
      \[
        |\det(M(\k_{1},\ldots,\k_{n}))| = \Norm_{K}(\mf{a})|\det(M(\a_{1},\ldots,\a_{n}))|,
      \]
      as claimed. The second identity follows from the first upon taking \(\mf{f} = \mc{O}_{K}\).
    \end{proof}
    
    As \(\s_{K}\) is also a \(\Q\)-embedding, \(\s_{K}(\mf{f})\) is a complete integral lattice of \(\R^{n}\). As a corollary of the previous result, we can determine the covolume of \(V_{\s_{K}(\mf{f})}\) of \(\s_{K}(\mf{f})\) as well.
    
    \begin{corollary}\label{cor:covolume_of_fractional_ideal_under_canonical_embedding}
      Let \(K\) be a number field with signature \((r_{1},r_{2})\) and let \(\mf{f}\) be a fractional ideal of \(K\). Then
      \[
        V_{\s_{K}(\mf{f})} = \Norm_{K}(\mf{f})\frac{\sqrt{|\D_{K}|}}{2^{r_{2}}},
      \]
      and
      \[
        V_{\s_{K}(\mc{O}_{K})} = \frac{\sqrt{|\D_{K}|}}{2^{r_{2}}}.
      \]
    \end{corollary}
    \begin{proof}
      Let \(\k_{1},\ldots,\k_{n}\) be a basis for \(K/\Q\) contained in \(\mf{f}\). Also let \(\rho_{1},\ldots,\rho_{r_{1}}\) and \(\tau_{1},\ldots,\tau_{r_{2}}\) run over the real \(\Q\)-embeddings and a complete set of representatives for the pairs of complex \(\Q\)-embeddings of \(\Hom_{\Q}(K,\conj{\Q})\) respectively. Then the associated generator matrix for \(\s_{K}(\mf{f})\) is
      \[
        \begin{pmatrix} \rho_{1}(\k_{1}) & \cdots & \rho_{r_{1}}(\k_{1}) & \Re(\tau_{1}(\k_{1})) & \Im(\tau_{1}(\k_{1})) & \cdots & \Re(\tau_{r_{2}}(\k_{1})) & \Im(\tau_{r_{2}}(\k_{1})) \\ \vdots & & \vdots & \vdots & \vdots & & \vdots & \vdots \\ \rho_{1}(\k_{n}) & \cdots & \rho_{r_{1}}(\k_{n}) & \Re(\tau_{1}(\k_{n})) & \Im(\tau_{1}(\k_{n})) & \cdots & \Re(\tau_{r_{2}}(\k_{n})) & \Im(\tau_{r_{2}}(\k_{n})) \end{pmatrix}^{t}.
      \]
      By \cref{prop:covolume_of_fractional_ideal_under_j} we are done if the absolute value of the determinant of this matrix is \(\frac{1}{2^{r_{2}}}\) times the absolute value of the determinate of the embedding matrix for \(\k_{1},\ldots,\k_{n}\). To show this, first add an \(i\) multiple of the imaginary columns to their corresponding real columns and then apply the identity \(\Im(z) = \frac{z-\conj{z}}{2i}\) to the imaginary columns. We obtain the matrix
      \[
        \begin{pmatrix} \rho_{1}(\k_{1}) & \cdots & \rho_{r_{1}}(\k_{1}) & \tau_{1}(\k_{1}) & \frac{\tau_{1}(\k_{1})-\conj{\tau_{1}}(\k_{1})}{2i} & \cdots & \tau_{r_{2}}(\k_{1}) & \frac{\tau_{r_{2}}(\k_{1})-\conj{\tau_{r_{2}}}(\k_{1})}{2i} \\ \vdots & & \vdots & \vdots & \vdots & & \vdots & \vdots \\ \rho_{1}(\k_{n}) & \cdots & \rho_{r_{1}}(\k_{n}) & \tau_{1}(\k_{n}) & \frac{\tau_{1}(\k_{n})-\conj{\tau_{1}}(\k_{n})}{2i} & \cdots & \tau_{r_{2}}(\k_{n}) & \frac{\tau_{r_{2}}(\k_{n})-\conj{\tau_{r_{2}}}(\k_{n})}{2i} \end{pmatrix}^{t},
      \]
      which has the same determinant as the generator matrix. Now multiply the imaginary columns by \(-2i\) and then adding the preceding columns to annihilate the negative terms. This results in the matrix
      \[
        \begin{pmatrix} \rho_{1}(\k_{1}) & \cdots & \rho_{r_{1}}(\k_{1}) & \tau_{1}(\k_{1}) &\conj{\tau_{1}}(\k_{1}) & \cdots & \tau_{r_{2}}(\k_{1}) & \conj{\tau_{r_{2}}}(\k_{1}) \\ \vdots & & \vdots & \vdots & \vdots & & \vdots & \vdots \\ \rho_{1}(\k_{n}) & \cdots & \rho_{r_{1}}(\k_{n}) & \tau_{1}(\k_{n}) & \conj{\tau_{1}}(\k_{n}) & \cdots & \tau_{r_{2}}(\k_{n}) & \conj{\tau_{r_{2}}}(\k_{n}) \end{pmatrix}^{t},
      \]
      whose determinant is \((-2i)^{-r_{2}}\) that of the generator matrix. But this latter matrix is the embedding \(M(\k_{1},\ldots,\k_{n})\). Whence
      \[
        V_{\s_{K}(\mf{f})} = \frac{|\det(M(\k_{1},\ldots,\k_{n}))|}{2^{r_{2}}}.
      \]
    \end{proof}
  \section{Finiteness of the Class Number}
    Recall that the class number of a Dedekind domain is a measure of how much the Dedekind domain fails to be a principal ideal domain and even need not be finite from \cref{rem:general_class_number_not_finite}. However, we will show that the class number \(h_{K}\) for a number field \(K\) is always finite and thus \(\mc{O}_{K}\) only has finite failure from being a principal ideal domain. In doing so, we will also show that every ideal class is represented by an integral ideal of bounded norm.

    \begin{theorem}\label{equ:finitness_of_class_number}
      Let \(K\) be a number field of degree \(n\) and signature \((r_{1},r_{2})\). Also, let \(X \subseteq \R^{n}\) be a compact convex symmetric set and set \(M = \max_{\mathbf{x} \in X}\left(\prod_{i}|x_{i}|\right)\). Then every ideal class of \(\Cl(K)\) is represented by an integral ideal \(\mf{a}\) satisfying
      \[
        \Norm_{K}(\mf{a}) \le \frac{2^{r_{1}+r_{2}}M}{\Vol(X)}\sqrt{|\D_{K}|}.
      \]
      Moreover, the ideal class group \(\Cl(K)\) is finite which is to say that the class number \(h_{K}\) is.
    \end{theorem}
    \begin{proof}
      Let \(\mf{f}\) be a fractional ideal, \(n\) be a positive integer, and set
      \[
        \l^{n} = 2^{n}\frac{V_{\s_{K}(\mf{f}^{-1})}}{\Vol(X)}.
      \]
      Then by construction, we have
      \[
        \Vol(\l X) = 2^{n}V_{\s_{K}(\mf{f}^{-1})}.
      \]
      By Minkowski's lattice point theorem (see \cref{sec:integral_lattices}), there exists a nonzero element of \(\s_{K}(\mf{f}^{-1}) \cap \l X\). Whence there is a nonzero \(\a \in \mf{f}^{-1}\) such that \(\s_{K}(\a) \in \l X\). Moreover, \(\a\mf{f} \subseteq \mc{O}_{K}\) so that \(\mf{a} = \a\mf{f}\) is an integral ideal in the same class as \(\mf{f}\). Using \cref{prop:norm_and_field_norms_are_identical,prop:formulas_for_trace_and_norm}, a short computation shows
      \[
        \Norm_{K}(\mf{a}) \le \l^{n}M\Norm_{K}(\mf{f}).
      \]
      In view of our choice of \(\l^{n}\) and \cref{cor:covolume_of_fractional_ideal_under_canonical_embedding}, this inequality becomes
      \[
        \Norm_{K}(\mf{a}) \le \frac{2^{r_{1}+r_{2}}M}{\Vol(X)}\sqrt{|\D_{K}|},
      \]
      which proves the first statement.
      
      We now prove that the class group is finite. By the first statement, we can find a complete set of representatives for \(\Cl(K)\) consisting of integral ideals of bounded norm. Since the norm is multiplicative, the prime factors of these representatives have bounded norm as well. Moreover, the norm of a prime \(\mf{p}\) is \(p^{f_{p}(\mf{p})}\) where \(p\) is the prime below \(\mf{p}\). Whence there are only finitely many possible primes \(p\). As there are finitely many primes \(\mf{p}\) above any prime \(p\), it follows that these representatives have finitely many prime factors. Altogether this means that there are finitely many representatives. Hence \(\Cl(K)\) is finite and so the class number \(h_{K}\) is too.
    \end{proof}

    Observe that the bound in the previous result is not explicit. To obtain an explicit bound, we need to make a choice for the compact convex symmetric set \(X\). In order for such a bound to be sharp, we need to ensure that the volume of \(X\) is large while the constant \(M\) is small. The following lemma dictates our choice of \(X\) and computes its volume:

    \begin{lemma}\label{lem:Minkowski_bound_lemma}
      Suppose \(n\) is a positive integer and write \(n = r_{1}+2r_{2}\) for some nonnegative integers \(r_{1}\) and \(r_{2}\). Let \(X \subset \R^{n}\) be the compact convex symmetric set given by
      \[
        X = \left\{(\mathbf{x}_{1},\mathbf{x}_{2}) \in \R^{n}:\sum_{i}|x_{i,1}|+2\sum_{j}\sqrt{x_{2j-1,2}^{2}+x_{2j,2}^{2}} \le n\right\}.
      \]
      Then
      \[
        \Vol(X) = \frac{n^{n}}{n!}2^{r_{1}}\left(\frac{\pi}{2}\right)^{r_{2}}.
      \]
    \end{lemma}
    \begin{proof}
      Making the change of variables \(x_{2j-1,2} \mapsto r_{j}\sin(\t_{j})\) and \(x_{2j,2} \mapsto r_{j}\cos(\t_{j})\) gives
      \[
        \Vol(X) = \int_{X'}r_{1} \cdots r_{r_{2}}\,d\mathbf{x}_{1}\,d\mathbf{r}\,d\boldsymbol{\t},
      \]
      because the Jacobian determinant is \(r_{1} \cdots r_{r_{2}}\), and where
      \[
        X' = \left\{(\mathbf{x}_{1},\mathbf{r},\boldsymbol{\t}) \in [-n,n]^{r_{1}} \op [0,n]^{r_{2}} \op [0,2\pi]^{r_{2}}:\sum_{i}|x_{i}|+2\sum_{j}r_{j} \le n\right\}.
      \]
      Making the change of variables \(r_{j} \mapsto \frac{r_{j}}{2}\) and using the facts that the integrand is symmetric in the \(x_{i}\) and independent of \(\t_{j}\) gives
      \[
        \Vol(X) = 2^{r_{1}}\left(\frac{\pi}{2}\right)^{r_{2}}\int_{X''}r_{1} \cdots r_{r_{2}}\,d\mathbf{x}_{1}\,d\mathbf{r},
      \]
      where
      \[
        X'' = \left\{(\mathbf{x}_{1},\mathbf{r}) \in [0,n]^{r_{1}+r_{2}}:\sum_{i}x_{i}+2\sum_{j}r_{j} \le n\right\}.
      \]
      To compute the remaining integral, for nonnegative integers \(\ell\) and \(k\) and positive \(t\), let
      \[
        X''_{\ell,k}(t) = \left\{(\mathbf{x}_{1},\mathbf{r}) \in [0,t]^{\ell+k}:\sum_{i}x_{i}+2\sum_{j}r_{j} \le t\right\},
      \]
      and define
      \[
        I_{\ell,k}(t) = \int_{X''_{\ell,k}(t)}r_{1} \cdots r_{\ell}\,d\mathbf{x}_{1}\,d\mathbf{r}.
      \]
      In this notation, we have
      \[
        \Vol(X) = 2^{r_{1}}\left(\frac{\pi}{2}\right)^{r_{2}}I_{r_{1},r_{2}}(n).
      \]
      To compute the remaining integral, the change of variables \(x_{i} \mapsto tx_{i}\) and \(r_{j} \mapsto tr_{j}\) gives
      \[
        I_{\ell,k}(t) = t^{\ell+2k}I_{\ell,k}(1).
      \]
      Whence
      \[
        I_{\ell,k}(1) = \int_{0}^{1}(1-x_{\ell})^{\ell-1+2k}I_{\ell-1,k}(1)\,dx_{\ell} = \frac{1}{\ell+2k}I_{\ell-1,k}(1).
      \]
      Upon repeating this procedure \(\ell-1\) times results in
      \[
        I_{\ell,k}(1) = \frac{1}{(\ell+2k) \cdots (2k+1)}I_{0,k}(1).
      \]
      Similarly, we have
      \[
        I_{0,k}(1) = \int_{0}^{1}r_{k}(1-r_{k})^{2k-2}I_{0,k-1}(1)\,dr_{k} = \frac{1}{2k}I_{0,k-1}(1),
      \]
      because the latter integral is the beta function \(B(1,2k-1)\) (see \cref{sec:the_gamma_function}). Upon repeating this procedure \(k-1\) times results in
      \[
        I_{0,k}(1) = \frac{1}{k!}.
      \]
      Altogether, this means
      \[
        I_{\ell,k}(t) = \frac{t^{\ell+2k}}{(\ell+2k)!},
      \]
      and therefore
      \[
        \Vol(X) = \frac{n^{n}}{n!}2^{r_{1}}\left(\frac{\pi}{2}\right)^{r_{2}}.
      \]
    \end{proof}

    Observe that the compact convex symmetric set \(X\) consists of those points in \(\R^{n}\) whose norm corresponding to the inner product induced by the Minkowski inner product from \cref{prop:Minkowski_space_isomorphism} at most \(n\). We can now obtain an explicit bound known as the \textbf{Minkowski bound}\index{Minkowski bound}.

    \begin{theorem*}[Minkowski bound]
      Let \(K\) be a number field of degree \(n\) and signature \((r_{1},r_{2})\). Then every ideal class of \(\Cl(K)\) is represented by an integral ideal \(\mf{a}\) satisfying
      \[
        \Norm_{K}(\mf{a}) \le \left(\frac{4}{\pi}\right)^{r_{2}}\frac{n!}{n^{n}}\sqrt{|\D_{K}|}.
      \]
    \end{theorem*}
    \begin{proof}
      \cref{equ:finitness_of_class_number,lem:Minkowski_bound_lemma} together show that every ideal class of \(\Cl(K)\) is represented by an integral ideal \(\mf{a}\) satisfying
      \[
        \Norm_{K}(\mf{a}) \le M\left(\frac{4}{\pi}\right)^{r_{2}}\frac{n!}{n^{n}}\sqrt{|\D_{K}|},
      \]
      where \(M = \max_{\mathbf{x} \in X}\left(\prod_{i}|x_{}|\right)\). But for all \(\mathbf{x} \in X\), the arithmetic-geometric mean inequality implies
      \[
        \left(\prod_{i}|x_{i}|\right)^{\frac{1}{n}} \le 1,
      \]
      Whence \(M \le 1\) completing the proof.
    \end{proof}

    As a corollary we can obtain a lower bound for the discriminant of a number field and show that every number field \(K\), other than \(\Q\), has at least one ramified prime.

    \begin{corollary}
      Let \(K\) be a number field of degree \(n\). Then
      \[
        |\D_{K}| \ge \left(\frac{\pi}{4}\right)^{\frac{n}{2}}\frac{n^{n}}{n!}.
      \]
      In particular, there is at least one ramified prime of \(K\) provided the degree of \(K\) is at least \(2\). 
    \end{corollary}
    \begin{proof}
      The inequality follows from the Minkowski bound since the norm of every integral ideal is at least \(1\) and \(r_{2}\) is at most \(n\). This proves the first statement. For the second statement, suppose \(n \ge 2\). In the case \(n = 2\), the lower bound is larger than \(1\) so that \(|\D_{K}|\) is at least \(2\). Whence \(|\D_{K}|\) has a prime divisor and so some prime above this divisor is a ramified prime of \(K\). Now suppose \(n \ge 3\). As \(n^{n} \ge n!\), \(\left(\frac{\pi}{4}\right)^{\frac{n}{2}}\frac{n^{n}}{n!}\) is an increasing function in \(n\). Therefore \(|\D_{K}| \ge 2\) and, as before, there is at least one ramified prime of \(K\) .
    \end{proof}

    Generally speaking, the class number of \(K\) is one of the most difficult pieces of arithmetic data about \(K\) to compute. For example, it is still unknown if there are infinitely many number fields of class number \(1\). That is to say, it is unknown if there are infinitely many number fields such that their ring of integers are principal ideal domains.

    Let \(K\) be a number field of degree \(n\) and signature \((r_{1},r_{2})\). We define the \textbf{rank}\index{rank} \(r_{K}\) of \(K\) by
    \[
      r_{K} = r_{1}+r_{2}-1.
    \]
    The rank of \(K\) is intimately related to the structure of the unit group \(\mc{O}_{K}^{\ast}\). Observe that we can express the unit group as
    \[
      \mc{O}_{K}^{\ast} = \{\a \in \mc{O}_{K}:\Norm_{K}(\a) = \pm 1\}.
    \]
    Our main goal will be to describe the group structure of \(\mc{O}_{K}^{\ast}\) completely. Let \(\mu(K)\) be the subgroup of \(K^{\ast}\) consisting of all the roots of unity in \(K\). As any \(m\)-th root of unity is a root of \(x^{m}-1\), it follows that roots of unity are algebraic integers. Whence \(\mu(K)\) is a subgroup of \(\mc{O}_{K}^{\ast}\). Clearly \(\{\pm 1\} \subseteq \mu(K)\). In fact, \(\mu(K)\) always is finite. To see this, let \(\w \in \mu(K)\) and recall that any root of unity is a primitive root of unity of some, potentially smaller, order. So we may assume \(\w\) is a primitive \(m\)-th root of unity. The minimal polynomial of \(\w\) over \(\Q\) is the \(m\)-th cyclotomic polynomial \(\Phi_{m}(x)\) which has degree \(\vphi(m)\). Whence \(\vphi(m) \le n\). Since \(\vphi(m) \to \infty\) as \(m \to \infty\), there can only be finitely many such \(\w\). This means \(\mu(K)\) is finite. Accordingly, set
    \[
      w_{K} = |\mu(K)|.
    \]
    Our primary aim will be to show that \(\mc{O}_{K}^{\ast}\) is a direct product of \(\mu(K)\) and a free abelian group of rank \(r_{K}\). Determining that the rank of the free abelian group is exactly \(r_{K}\) will be the most difficult part of the argument. To accomplish this, we will require a map on \(K^{\ast}\) that transitions between the Minkowski trace and Minkowski norm. To define this map, let \(\rho\) and \(\tau\) run over the real \(\Q\)-embeddings and a complete set of representatives for the pairs of complex \(\Q\)-embeddings of \(\Hom_{\Q}(K,\conj{\Q})\) respectively.     \[

    \[
      \Trace_{K_{\R}}(\ell(\mathbf{z})) = \log|\Norm_{K_{\R}}(\mathbf{z})|,

      \log_{K}:K^{\ast} \to \R^{r_{K}+1} \qquad \k \mapsto ((\log|\rho(\k)|)_{\rho},(2\log|\rho(\tau)|)_{\tau}).

    \[
      U = \{\mathbf{z} \in K_{\R}^{\ast}:\Norm_{K_{\R}}(\mathbf{z}) = \pm 1\}.

    \[

    \begin{proposition}\label{prop:exact_sequence_for_Dirichlets_unit_theorem}
      Let \(K\) be a number field. Then there is a short exact sequence

      \begin{center}
        \begin{tikzcd}
          1 \arrow{r} & \mu(K) \arrow{r} & \mc{O}_{K}^{\ast} \arrow{r} & \L_{K} \arrow{r} & 0,
        \end{tikzcd}
      \end{center}
      induced by natural inclusions and where the third map is given by \(\l\).
    \end{proposition}
    \begin{proof}
      As the second map is injective and the third map is surjective, the sequence is exact at \(\mu(K)\) and \(\L_{K}\). So it remains to prove exactness at \(\mc{O}_{K}^{\ast}\). For this we must show \(\ker\l = \mu(K)\). The reverse inclusion holds since the elements of \(\Hom_{\Q}(K,\conj{\Q})\) preserve roots of unity. For the forward inclusion, the elements of \(j(\ker\l)\) are such that each component is absolutely bounded. So one the one hand, this set belongs to a compact subset of \(K_{\R}\). On the other hand, this set is also a subset of the lattice \(j(\mc{O}_{K})\) which is a discrete subset of \(K_{\R}\). Together, this means \(j(\ker\l)\) is a subset of a compact and discrete set which is necessarily finite. As \(j\) is a \(\Q\)-embedding, it follows that the subgroup \(\ker\l\) of \(\mc{O}_{K}^{\ast}\) contains finitely many elements and hence only roots of unity since these are the only elements of \(\C\) with finite order. This proves the forward inclusion.
    \end{proof}

    Our aim now is to show that the unit lattice \(\L_{K}\) is a a free abelian group of rank \(r_{K}\). For this, we will require a lemma.

    \begin{lemma}\label{lem:finitely_many_elements_of_fixed_norm}
      Let \(K\) be a number field. There are finitely many elements in \(\mc{O}_{K}\) of a given field norm up to multiplication by units.
    \end{lemma}
    \begin{proof}
      The claim is true for algebraic integers of norm \(\pm 1\) as these are exactly the units of \(\mc{O}_{K}\). Now suppose \(m \ge 2\). By \cref{prop:norm_and_field_norms_are_identical}, \(\mc{O}_{K}/m\mc{O}_{K}\) is finite with \(\Norm_{K}(m)\) many elements. Therefore, it suffices to show that in each coset there is at most finitely many element of norm \(n\) up to multiplication by units. In fact, we will show that there is at most one such element. So let \(\a\) and \(\b\) be two representatives in the same coset which are of field norm \(n\). Writing \(\a = \b+n\g\) for some \(\g \in \mc{O}_{K}\), we may further write
      \[
        \frac{\a}{\b} = 1+\frac{\Norm_{K}(\b)}{\b}\g.
      \]
      As any integral ideal contains its norm, \cref{prop:norm_and_field_norms_are_identical} implies \(\frac{\Norm_{K}(\b)}{\b} \in \mc{O}_{K}\). Whence \(\frac{\a}{\b} \in \mc{O}_{K}\) and interchanging the roles of \(\a\) and \(\b\) shows that \(\frac{\b}{\a} \in \mc{O}_{K}\) too. But then \(\frac{\a}{\b}\) is a unit in \(\mc{O}_{K}\) and thus \(\a\) and \(\b\) differ up to multiplication by a unit.
    \end{proof}

    \[
      P = \begin{pmatrix} \frac{1}{\sqrt{r_{K}+1}} & \l(\e_{1})_{1} & \cdots & \l(\e_{r_{K}})_{1} \\ \vdots & \vdots & & \vdots \\ \frac{1}{\sqrt{r_{K}+1}} & \l(\e_{1})_{r_{K}+1} & \cdots & \l(\e_{r_{K}})_{r_{K}+1} \end{pmatrix}.

      P' = \begin{pmatrix} \frac{1}{\sqrt{r_{K}+1}} & \l(\e_{1})_{1} & \cdots & \l(\e_{r_{K}})_{1} \\ \vdots & \vdots & & \vdots \\ \sqrt{r_{K}+1} & 0 & & 0 \\ \vdots & \vdots & & \vdots \\ \frac{1}{\sqrt{r_{K}+1}} & \l(\e_{1})_{r_{K}+1} & \cdots & \l(\e_{r_{K}})_{r_{K}+1} \end{pmatrix},
